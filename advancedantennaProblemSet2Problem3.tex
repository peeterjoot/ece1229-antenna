\makeoproblem
%{Far field electric field away from sources.}
{Far field electric field.}
{advancedantenna:problemSet2:3}{2015 ps2, p3}{
\index{far field!electric field}
%
Show that in a region of space where there are no sources, the electric field derived from the magnetic vector potential is given by the expression:
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:20}
\BE = \inv{j \omega \epsilon_0 \mu_0 } \spacegrad \cross \lr{\spacegrad \cross \BA}.
\end{equation}
\index{electric far field}
} % makeoproblem
%
\makeanswer{advancedantenna:problemSet2:3}{
%
First consider the expansion of the curls
\index{potential!curl of curl}
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:40}
\begin{aligned}
\spacegrad \cross \lr{ \spacegrad \cross \BA }
&=
\epsilon_{rst} \Be_r \partial_s \epsilon_{tbc} \partial_b A_c
\\ &=
\delta_{rs}^{[bc]} \Be_r \partial_s \partial_b A_c
\\ &=
\Be_r \partial_s \lr{ \partial_r A_s - \partial_s A_r },
\end{aligned}
\end{equation}
%
so
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:60}
\spacegrad \cross \lr{ \spacegrad \cross \BA }
= \spacegrad \lr{ \spacegrad \cdot A} - \spacegrad^2 \BA.
\end{equation}
%
This supplies a strong hint of how to proceed.  The electric field can be expanded utilizing this relation and the Helmholtz equation relating \( \BA \) and \( \BJ \)
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:80}
\begin{aligned}
\BE
&= -j \omega \BA - j \inv{ \omega \epsilon_0 \mu_0 } \spacegrad \lr{ \spacegrad \cdot \BA }
\\ &= -j \omega \BA - j \inv{ \omega \epsilon_0 \mu_0 } \lr{
\spacegrad \cross \lr{ \spacegrad \cross \BA } + \spacegrad^2 \BA
}
\\ &= -j \omega \BA - j \inv{ \omega \epsilon_0 \mu_0 } \lr{
\spacegrad \cross \lr{ \spacegrad \cross \BA } - \lr{ k^2 \BA + \mu_0 \BJ }
}.
\end{aligned}
\end{equation}
%
However,
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:100}
\frac{k^2}{\omega \epsilon_0 \mu_0}
=
\frac{\omega^2/c^2}{\omega (1/c^2)}
=
\omega,
\end{equation}
%
leaving
%
\begin{equation}\label{eqn:advancedantennaProblemSet2Problem3:90}
\BE
=
- j \inv{ \omega \epsilon_0 \mu_0 }
\spacegrad \cross \lr{ \spacegrad \cross \BA }
+ j \inv{ \omega \epsilon_0 } \BJ.
\end{equation}
%
Since \( \BJ = 0 \) in a region of space where there are no sources, the electric field in such a region is given by \cref{eqn:advancedantennaProblemSet2Problem3:20} as stated.
}
